public static void main(String[] args) {
		try {
			ObjectMapper om = new ObjectMapper();
			SummonerVO svo = null;
			HttpClient hc = HttpClientBuilder.create().build();
			HttpGet hg = new HttpGet(
					"https://kr.api.riotgames.com/lol/summoner/v4/summoners/by-name/우솝집사작살?api_key=RGAPI-5896d653-ad14-4c12-88c9-c9135eb97d74");
			HttpResponse hr = hc.execute(hg);
			if(hr.getStatusLine().getStatusCode() == 200) {
				ResponseHandler<String> h = new BasicResponseHandler();
				String body = h.handleResponse(hr);
				svo = om.readValue(body, SummonerVO.class);
				System.out.println(svo.getName());
				System.out.println(svo.getProfileIconId());
			}
		} catch (Exception e) {
			System.out.println(e);
		}

	}
